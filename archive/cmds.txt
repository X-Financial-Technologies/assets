# Create empty file
touch filename.txt

# Create file with content
echo "some content" > filename.txt    # Overwrites if file exists
echo "more content" >> filename.txt   # Appends to file

# Create and edit file with nano editor
nano filename.txt

# Create and edit with vim
vim filename.txt

# Create file with heredoc
cat << EOF > filename.txt
line 1
line 2
line 3
EOF

# Create multiple empty files
touch file1.txt file2.txt file3.txt
